pub const DEFAULT_BOOL_FORMATTER: BoolFormatter<'_>;
Expand description

String formatter used by default in Confirm prompts. Translates bool to "Yes" and false to "No".

ยงExamples

use inquire::formatter::DEFAULT_BOOL_FORMATTER;

let formatter = DEFAULT_BOOL_FORMATTER;
assert_eq!(String::from("Yes"), formatter(true));
assert_eq!(String::from("No"), formatter(false));
