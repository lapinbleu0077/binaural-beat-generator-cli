5/// Type alias to represent the function used to Score and filter options.
6///
7/// The function receives:
8/// - Current user input, filter value
9/// - Current option being evaluated, with type preserved
10/// - String value of the current option
11/// - Index of the current option in the original list
12///
13/// The return type should be a score determining the order options should be displayed to the user.
14/// The greater the score, the higher on the list it will be displayed.
15///
16/// # Examples
17///
18/// ```
19/// use inquire::type_aliases::Scorer;
20///
21/// // Implement simpler 'contains' filter that maintains the current order
22/// // If all scores are the same, no sorting will occur
23/// let filter: Scorer<str> =
24///     &|input, _option, string_value, _idx| -> Option<i64> {
25///        let filter = input.to_lowercase();
26///        match string_value.to_lowercase().contains(&filter) {
27///            true => Some(0),
28///            false => None,
29///        }
30///     };
31///
32/// assert_eq!(None, filter("sa", "New York",      "New York",      0));
33/// assert_eq!(None, filter("sa", "Los Angeles",   "Los Angeles",   1));
34/// assert_eq!(None, filter("sa", "Chicago",       "Chicago",       2));
35/// assert_eq!(None, filter("sa", "Houston",       "Houston",       3));
36/// assert_eq!(None, filter("sa", "Phoenix",       "Phoenix",       4));
37/// assert_eq!(None, filter("sa", "Philadelphia",  "Philadelphia",  5));
38/// assert_eq!(Some(0), filter("sa", "San Antonio",   "San Antonio",   6));
39/// assert_eq!(Some(0), filter("sa", "San Diego",     "San Diego",     7));
40/// assert_eq!(None, filter("sa", "Dallas",        "Dallas",        8));
41/// assert_eq!(Some(0), filter("sa", "San Francisco", "San Francisco", 9));
42/// assert_eq!(None, filter("sa", "Austin",        "Austin",       10));
43/// assert_eq!(None, filter("sa", "Jacksonville",  "Jacksonville", 11));
44/// assert_eq!(Some(0), filter("sa", "San Jose",      "San Jose",     12));
45///```
46///
47///
48///
49/// Default implementation for fuzzy search (almost)
50///```
51/// use inquire::type_aliases::Scorer;
52/// use fuzzy_matcher::{skim::SkimMatcherV2, FuzzyMatcher};
53///
54/// pub const DEFAULT_SCORER: Scorer<str> =
55///     &|input, _option, string_value, _idx| -> Option<i64> {
56///         let matcher = SkimMatcherV2::default().ignore_case();
57///         matcher.fuzzy_match(string_value, input)
58///     };
59///
60/// assert_eq!(None,     DEFAULT_SCORER("sa", &"New York",      "New York",      0));
61/// assert_eq!(Some(49), DEFAULT_SCORER("sa", &"Sacramento",    "Sacramento",    1));
62/// assert_eq!(Some(35), DEFAULT_SCORER("sa", &"Kansas",        "Kansas",        2));
63/// assert_eq!(Some(35), DEFAULT_SCORER("sa", &"Mesa",          "Mesa",          3));
64/// assert_eq!(None,     DEFAULT_SCORER("sa", &"Phoenix",       "Phoenix",       4));
65/// assert_eq!(None,     DEFAULT_SCORER("sa", &"Philadelphia",  "Philadelphia",  5));
66/// assert_eq!(Some(49), DEFAULT_SCORER("sa", &"San Antonio",   "San Antonio",   6));
67/// assert_eq!(Some(49), DEFAULT_SCORER("sa", &"San Diego",     "San Diego",     7));
68/// assert_eq!(None,     DEFAULT_SCORER("sa", &"Dallas",        "Dallas",        8));
69/// assert_eq!(Some(49), DEFAULT_SCORER("sa", &"San Francisco", "San Francisco", 9));
70/// assert_eq!(None,     DEFAULT_SCORER("sa", &"Austin",        "Austin",        10));
71/// assert_eq!(None,     DEFAULT_SCORER("sa", &"Jacksonville",  "Jacksonville",  11));
72/// assert_eq!(Some(49), DEFAULT_SCORER("sa", &"San Jose",      "San Jose",      12));
73/// ```
74pub type Scorer<'a, T> = &'a dyn Fn(&str, &T, &str, usize) -> Option<i64>;

76/// Type alias to represent the function used to retrieve text input suggestions.
77/// The function receives the current input and should return a collection of strings
78/// containing the suggestions to be made to the user.
79pub type Suggester<'a> = &'a dyn Fn(&str) -> Result<Vec<String>, CustomUserError>;
80
81/// Type alias to represent the function used to retrieve an optional autocompletion suggestion.
82/// The function receives the current input and should return the suggestion (if any)
83/// that will replace the current input.
84pub type Completer<'a> = &'a dyn Fn(&str) -> Result<Option<String>, CustomUserError>;
