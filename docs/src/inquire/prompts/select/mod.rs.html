30/// Prompt suitable for when you need the user to select one option among many.
31///
32/// The user can select and submit the current highlighted option by pressing enter.
33///
34/// This prompt requires a prompt message and a **non-empty** `Vec` of options to be displayed to the user. The options can be of any type as long as they implement the `Display` trait. It is required that the `Vec` is moved to the prompt, as the prompt will return the selected option (`Vec` element) after the user submits.
35/// - If the list is empty, the prompt operation will fail with an `InquireError::InvalidConfiguration` error.
36///
37/// This prompt does not support custom validators because of its nature. A submission always selects exactly one of the options. If this option was not supposed to be selected or is invalid in some way, it probably should not be included in the options list.
38///
39/// The options are paginated in order to provide a smooth experience to the user, with the default page size being 7. The user can move from the options and the pages will be updated accordingly, including moving from the last to the first options (or vice-versa).
40///
41/// Like all others, this prompt also allows you to customize several aspects of it:
42///
43/// - **Prompt message**: Required when creating the prompt.
44/// - **Options list**: Options displayed to the user. Must be **non-empty**.
45/// - **Starting cursor**: Index of the cursor when the prompt is first rendered. Default is 0 (first option). If the index is out-of-range of the option list, the prompt will fail with an [`InquireError::InvalidConfiguration`] error.
46/// - **Starting filter input**: Sets the initial value of the filter section of the prompt.
47/// - **Help message**: Message displayed at the line below the prompt.
48/// - **Formatter**: Custom formatter in case you need to pre-process the user input before showing it as the final answer.
49///   - Prints the selected option string value by default.
50/// - **Page size**: Number of options displayed at once, 7 by default.
51/// - **Display option indexes**: On long lists, it might be helpful to display the indexes of the options to the user. Via the `RenderConfig`, you can set the display mode of the indexes as a prefix of an option. The default configuration is `None`, to not render any index when displaying the options.
52/// - **Scorer function**: Function that defines the order of options and if displayed as all.
53///
54/// # Example
55///
56/// ```no_run
57/// use inquire::{error::InquireError, Select};
58///
59/// let options: Vec<&str> = vec!["Banana", "Apple", "Strawberry", "Grapes",
60///     "Lemon", "Tangerine", "Watermelon", "Orange", "Pear", "Avocado", "Pineapple",
61/// ];
62///
63/// let ans: Result<&str, InquireError> = Select::new("What's your favorite fruit?", options).prompt();
64///
65/// match ans {
66///     Ok(choice) => println!("{}! That's mine too!", choice),
67///     Err(_) => println!("There was an error, please try again"),
68/// }
69/// ```
70///
71/// [`InquireError::InvalidConfiguration`]: crate::error::InquireError::InvalidConfiguration
72#[derive(Clone)]
73pub struct Select<'a, T> {
74    /// Message to be presented to the user.
75    pub message: &'a str,
76
77    /// Options displayed to the user.
78    pub options: Vec<T>,
79
80    /// Help message to be presented to the user.
81    pub help_message: Option<&'a str>,
82
83    /// Page size of the options displayed to the user.
84    pub page_size: usize,
85
86    /// Whether vim mode is enabled. When enabled, the user can
87    /// navigate through the options using hjkl.
88    pub vim_mode: bool,
89
90    /// Starting cursor index of the selection.
91    pub starting_cursor: usize,
92
93    /// Starting filter input
94    pub starting_filter_input: Option<&'a str>,
95
96    /// Reset cursor position to first option on filter input change.
97    /// Defaults to true.
98    pub reset_cursor: bool,
99
100    /// Whether to allow the option list to be filtered by user input or not.
101    ///
102    /// Defaults to true.
103    pub filter_input_enabled: bool,
104
105    /// Function called with the current user input to score the provided
106    /// options.
107    pub scorer: Scorer<'a, T>,
108
109    /// Function that formats the user input and presents it to the user as the final rendering of the prompt.
110    pub formatter: OptionFormatter<'a, T>,
111
112    /// RenderConfig to apply to the rendered interface.
113    ///
114    /// Note: The default render config considers if the NO_COLOR environment variable
115    /// is set to decide whether to render the colored config or the empty one.
116    ///
117    /// When overriding the config in a prompt, NO_COLOR is no longer considered and your
118    /// config is treated as the only source of truth. If you want to customize colors
119    /// and still support NO_COLOR, you will have to do this on your end.
120    pub render_config: RenderConfig<'a>,
121}

127    /// String formatter used by default in [Select](crate::Select) prompts.
128    /// Simply prints the string value contained in the selected option.
129    ///
130    /// # Examples
131    ///
132    /// ```
133    /// use inquire::list_option::ListOption;
134    /// use inquire::Select;
135    ///
136    /// let formatter = Select::<&str>::DEFAULT_FORMATTER;
137    /// assert_eq!(String::from("First option"), formatter(ListOption::new(0, &"First option")));
138    /// assert_eq!(String::from("First option"), formatter(ListOption::new(11, &"First option")));
139    /// ```
140    pub const DEFAULT_FORMATTER: OptionFormatter<'a, T> = &|ans| ans.to_string();
141
142    /// Default scoring function, which will create a score for the current option using the input value.
143    /// The return will be sorted in Descending order, leaving options with None as a score.
144    ///
145    /// # Examples
146    ///
147    /// ```
148    /// use inquire::Select;
149    ///
150    /// let scorer = Select::<&str>::DEFAULT_SCORER;
151    /// assert_eq!(None,     scorer("sa", &"New York",      "New York",      0));
152    /// assert_eq!(Some(49), scorer("sa", &"Sacramento",    "Sacramento",    1));
153    /// assert_eq!(Some(35), scorer("sa", &"Kansas",        "Kansas",        2));
154    /// assert_eq!(Some(35), scorer("sa", &"Mesa",          "Mesa",          3));
155    /// assert_eq!(None,     scorer("sa", &"Phoenix",       "Phoenix",       4));
156    /// assert_eq!(None,     scorer("sa", &"Philadelphia",  "Philadelphia",  5));
157    /// assert_eq!(Some(49), scorer("sa", &"San Antonio",   "San Antonio",   6));
158    /// assert_eq!(Some(49), scorer("sa", &"San Diego",     "San Diego",     7));
159    /// assert_eq!(None,     scorer("sa", &"Dallas",        "Dallas",        8));
160    /// assert_eq!(Some(49), scorer("sa", &"San Francisco", "San Francisco", 9));
161    /// assert_eq!(None,     scorer("sa", &"Austin",        "Austin",        10));
162    /// assert_eq!(None,     scorer("sa", &"Jacksonville",  "Jacksonville",  11));
163    /// assert_eq!(Some(49), scorer("sa", &"San Jose",      "San Jose",      12));
164    /// ```
165    #[cfg(feature = "fuzzy")]
166    pub const DEFAULT_SCORER: Scorer<'a, T> =
167        &|input, _option, string_value, _idx| -> Option<i64> {
168            DEFAULT_MATCHER.fuzzy_match(string_value, input)
169        };
170
171    #[cfg(not(feature = "fuzzy"))]
172    pub const DEFAULT_SCORER: Scorer<'a, T> =
173        &|input, _option, string_value, _idx| -> Option<i64> {
174            let filter = input.to_lowercase();
175            match string_value.to_lowercase().contains(&filter) {
176                true => Some(0),
177                false => None,
178            }
179        };

303    /// Parses the provided behavioral and rendering options and prompts
304    /// the CLI user for input according to the defined rules.
305    ///
306    /// Returns the owned object selected by the user.
307    pub fn prompt(self) -> InquireResult<T> {
308        self.raw_prompt().map(|op| op.value)
309    }
310
311    /// Parses the provided behavioral and rendering options and prompts
312    /// the CLI user for input according to the defined rules.
313    ///
314    /// This method is intended for flows where the user skipping/cancelling
315    /// the prompt - by pressing ESC - is considered normal behavior. In this case,
316    /// it does not return `Err(InquireError::OperationCanceled)`, but `Ok(None)`.
317    ///
318    /// Meanwhile, if the user does submit an answer, the method wraps the return
319    /// type with `Some`.
320    pub fn prompt_skippable(self) -> InquireResult<Option<T>> {
321        match self.prompt() {
322            Ok(answer) => Ok(Some(answer)),
323            Err(InquireError::OperationCanceled) => Ok(None),
324            Err(err) => Err(err),
325        }
326    }
327
328    /// Parses the provided behavioral and rendering options and prompts
329    /// the CLI user for input according to the defined rules.
330    ///
331    /// Returns a [`ListOption`](crate::list_option::ListOption) containing
332    /// the index of the selection and the owned object selected by the user.
333    pub fn raw_prompt(self) -> InquireResult<ListOption<T>> {
334        let (input_reader, terminal) = get_default_terminal()?;
335        let mut backend = Backend::new(input_reader, terminal, self.render_config)?;
336        self.prompt_with_backend(&mut backend)
337    }
338
339    pub(crate) fn prompt_with_backend<B: SelectBackend>(
340        self,
341        backend: &mut B,
342    ) -> InquireResult<ListOption<T>> {
343        SelectPrompt::new(self)?.prompt(backend)
344    }
