34#[derive(Debug)]
35struct FrameState {
36    /// terminal size when the frame was rendered
37    pub terminal_size: TerminalSize,
38    /// resulting frame size
39    pub frame_size: Dimension,
40    /// position to put cursor after writing all present content
41    pub expected_cursor_position: Option<Position>,
42    /// content and pre-calculated hashes for each rendered line
43    /// the length of this vector should be equal to frame_size.height
44    pub finished_rows: Vec<FrameRow>,
45    pub current_styled: Styled<String>,
46    pub current_line: Vec<Styled<String>>,
47    pub current_line_width: u16,
48    pub current_line_hasher: FxHasher,
49}
50
51impl FrameState {
52    pub fn new(terminal_size: TerminalSize) -> Self {
53        Self {
54            terminal_size,
55            frame_size: Dimension::new(0, 0),
56            finished_rows: Vec::new(),
57            current_styled: Styled::default(),
58            current_line: Vec::new(),
59            current_line_hasher: FxHasher::default(),
60            current_line_width: 0,
61            expected_cursor_position: None,
62        }
63    }
64
65    pub fn write(&mut self, value: &Styled<impl AsRef<str> + Display>) {
66        self.current_styled.style = value.style;
67
68        for piece in value.content.ansi_aware_chars() {
69            piece.hash(&mut self.current_line_hasher);
70            value.style.hash(&mut self.current_line_hasher);
71
72            let current_char = match piece {
73                AnsiAwareChar::Char(c) => c,
74                AnsiAwareChar::AnsiEscapeSequence(seq) => {
75                    // we don't care for escape sequences when calculating cursor position
76                    // and box size
77                    self.current_styled.content.push_str(seq);
78                    continue;
79                }
80            };
81
82            if current_char == '\n' {
83                self.finish_line();
84                continue;
85            }
86
87            let remaining_width_space = self.terminal_size.width() - self.current_line_width;
88            let character_length = UnicodeWidthChar::width(current_char).unwrap_or(0) as u16;
89
90            if character_length > remaining_width_space {
91                // the character will (probably) not fit into the current line
92                self.finish_line();
93            }
94
95            self.current_line_width = self.current_line_width.saturating_add(character_length);
96            self.current_styled.content.push(current_char);
97        }
98
99        if !self.current_styled.content.is_empty() {
100            self.current_line
101                .push(std::mem::take(&mut self.current_styled));
102        }
103    }
104
105    pub fn mark_cursor_position(&mut self, offset: isize) {
106        let mut row = self.finished_rows.len() as u16;
107        let mut col = self.current_line_width;
108
109        col = col.saturating_add(offset as u16);
110
111        if col >= self.terminal_size.width() {
112            col -= self.terminal_size.width();
113            row += 1;
114        }
115
116        self.expected_cursor_position = Some(Position { row, col });
117    }
118
119    pub fn finish(&mut self) {
120        self.finish_line();
121    }
122
123    pub fn resize_if_needed(&mut self, new_size: TerminalSize) {
124        if new_size == self.terminal_size {
125            return;
126        }
127
128        let mut new_state = Self::new(new_size);
129        for row in &self.finished_rows {
130            for styled in row.get_content() {
131                new_state.write(styled);
132            }
133            new_state.finish_line();
134        }
135        for styled in &self.current_line {
136            new_state.write(styled);
137        }
138        new_state.finish_line();
139
140        *self = new_state;
141    }
142
143    fn finish_line(&mut self) {
144        let current_styled = std::mem::take(&mut self.current_styled);
145        self.current_styled.style = current_styled.style;
146
147        if !current_styled.content.is_empty() || !current_styled.style.is_empty() {
148            self.current_line.push(current_styled);
149        }
150
151        let hasher = std::mem::take(&mut self.current_line_hasher);
152        let content = std::mem::take(&mut self.current_line);
153
154        if content.is_empty() {
155            return;
156        }
157
158        self.finished_rows
159            .push(FrameRow::new(content, hasher.finish()));
160
161        self.frame_size = Dimension::new(
162            self.frame_size.width().max(self.current_line_width),
163            self.finished_rows.len() as u16,
164        );
165
166        if !self.current_styled.style.is_empty() {
167            self.current_styled
168                .style
169                .hash(&mut self.current_line_hasher);
170        }
171
172        self.current_line_width = 0;
173    }
174}

176#[derive(Debug, Default)]
177enum RenderState {
178    #[default]
179    Initial,
180    ActiveRender {
181        last_rendered_frame: FrameState,
182        current_frame: FrameState,
183    },
184    Rendered(FrameState),
185}
186
187pub struct FrameRenderer<T>
188where
189    T: Terminal,
190{
191    terminal: T,
192    cursor_position: Position,
193    state: RenderState,
194}
195
196impl<T> FrameRenderer<T>
197where
198    T: Terminal,
199{
200    pub fn new(terminal: T) -> io::Result<Self> {
201        Ok(Self {
202            terminal,
203            cursor_position: Position::default(),
204            state: RenderState::Initial,
205        })
206    }
207
208    pub fn write(&mut self, value: impl Display) -> io::Result<()> {
209        self.write_styled(Styled::new(value))
210    }
211
212    pub fn write_styled(&mut self, value: Styled<impl Display>) -> io::Result<()> {
213        match &mut self.state {
214            RenderState::Rendered(_) | RenderState::Initial => {}
215            RenderState::ActiveRender { current_frame, .. } => {
216                // here we are converting from a generic impl Display to String
217                // because we are storing the string content in the frame (we can't store a ref to an object, for example).
218                //
219                // we pay a little bit in memory/cpu usage for this so we can
220                // calculate incremental rendering and cursor position on-the-fly.
221                let formatted = format!("{}", value.content);
222                let value = value.with_content(formatted);
223
224                current_frame.write(&value);
225            }
226        }
227
228        Ok(())
229    }
230
231    pub fn mark_cursor_position(&mut self, offset: isize) {
232        match &mut self.state {
233            RenderState::Rendered(_) | RenderState::Initial => {}
234            RenderState::ActiveRender { current_frame, .. } => {
235                current_frame.mark_cursor_position(offset);
236            }
237        }
238    }
239
240    pub fn start_frame(&mut self) -> io::Result<()> {
241        let terminal_size = self.refresh_terminal_size();
242
243        self.state = match std::mem::replace(&mut self.state, RenderState::Initial) {
244            RenderState::Initial => RenderState::ActiveRender {
245                last_rendered_frame: FrameState::new(terminal_size),
246                current_frame: FrameState::new(terminal_size),
247            },
248
249            RenderState::Rendered(last_rendered_frame) => RenderState::ActiveRender {
250                last_rendered_frame,
251                current_frame: FrameState::new(terminal_size),
252            },
253
254            RenderState::ActiveRender {
255                last_rendered_frame,
256                current_frame,
257            } => RenderState::ActiveRender {
258                last_rendered_frame,
259                current_frame,
260            },
261        };
262
263        Ok(())
264    }
265
266    pub fn finish_current_frame(&mut self, add_empty_line: bool) -> io::Result<()> {
267        let (last_rendered_frame, mut current_frame) = match std::mem::take(&mut self.state) {
268            RenderState::Rendered(_) | RenderState::Initial => {
269                return Ok(());
270            }
271            RenderState::ActiveRender {
272                last_rendered_frame,
273                current_frame,
274            } => (last_rendered_frame, current_frame),
275        };
276
277        current_frame.finish();
278
279        let rows_to_iterate = std::cmp::max(
280            last_rendered_frame.frame_size.height(),
281            current_frame.frame_size.height(),
282        );
283
284        self.terminal.cursor_hide()?;
285        self.move_cursor_to(Position { row: 0, col: 0 })?;
286
287        for i in 0..rows_to_iterate {
288            let last_row = last_rendered_frame.finished_rows.get(i as usize);
289            let current_row = current_frame.finished_rows.get(i as usize);
290
291            match (last_row, current_row) {
292                (Some(last_row), Some(current_row)) => {
293                    if last_row.hash() != current_row.hash() {
294                        for styled in current_row.get_content() {
295                            self.terminal.write_styled(styled)?;
296                        }
297                        self.terminal.clear_until_new_line()?;
298                    }
299                }
300                (Some(_), None) => {
301                    self.terminal.clear_line()?;
302                }
303                (None, Some(current_row)) => {
304                    for styled in current_row.get_content() {
305                        self.terminal.write_styled(styled)?;
306                    }
307                }
308                (None, None) => {
309                    // unreachable, but we don't want to panic live :)
310                    #[cfg(test)]
311                    unreachable!(
312                        "frame_size should never be larger then finished_rows for both frames"
313                    )
314                }
315            }
316
317            self.terminal.write("\r")?;
318            self.cursor_position.col = 0;
319            if i + 1 < rows_to_iterate {
320                self.terminal.write("\n")?;
321                self.cursor_position.row += 1;
322            }
323        }
324
325        if add_empty_line {
326            self.terminal.write("\n")?;
327            self.cursor_position.row += 1;
328        }
329
330        if let Some(expected_cursor_position) = current_frame.expected_cursor_position {
331            self.move_cursor_to(expected_cursor_position)?;
332        }
333
334        self.terminal.cursor_show()?;
335        self.terminal.flush()?;
336
337        self.state = RenderState::Rendered(current_frame);
338
339        Ok(())
340    }
341
342    fn move_cursor_to_end_position(&mut self) -> io::Result<()> {
343        self.refresh_terminal_size();
344
345        let last_rendered = match &mut self.state {
346            RenderState::Initial => return Ok(()),
347            RenderState::ActiveRender {
348                last_rendered_frame,
349                ..
350            }
351            | RenderState::Rendered(last_rendered_frame) => last_rendered_frame,
352        };
353
354        let end_position = Position {
355            col: 0,
356            row: last_rendered.frame_size.height(),
357        };
358
359        self.move_cursor_to(end_position)?;
360
361        Ok(())
362    }
363
364    fn move_cursor_to(&mut self, position: Position) -> io::Result<()> {
365        let current_cursor_position = self.cursor_position;
366
367        match current_cursor_position.row.cmp(&position.row) {
368            Ordering::Greater => {
369                self.terminal
370                    .cursor_up(current_cursor_position.row - position.row)?;
371            }
372            Ordering::Less => {
373                self.terminal
374                    .cursor_down(position.row - current_cursor_position.row)?;
375            }
376            Ordering::Equal => {}
377        }
378
379        match current_cursor_position.col.cmp(&position.col) {
380            Ordering::Greater => {
381                self.terminal
382                    .cursor_left(current_cursor_position.col - position.col)?;
383            }
384            Ordering::Less => {
385                self.terminal
386                    .cursor_right(position.col - current_cursor_position.col)?;
387            }
388            Ordering::Equal => {}
389        }
390
391        self.cursor_position = position;
392
393        Ok(())
394    }
395
396    fn refresh_terminal_size(&mut self) -> TerminalSize {
397        // not properly handling resizes is better than panicking, so when
398        // getting the terminal size fails, we assume we're on a terminal
399        // that will always have enough space
400        let terminal_size = self
401            .terminal
402            .get_size()
403            .unwrap_or(TerminalSize::new(1000, 1000));
404
405        if terminal_size.width() < self.cursor_position.col {
406            let new_line_offset = self.cursor_position.col / terminal_size.width();
407            let new_col = self.cursor_position.col % terminal_size.width();
408            self.cursor_position = Position {
409                row: self.cursor_position.row + new_line_offset,
410                col: new_col,
411            };
412        }
413
414        match &mut self.state {
415            RenderState::Initial => {}
416            RenderState::ActiveRender {
417                current_frame,
418                last_rendered_frame,
419            } => {
420                last_rendered_frame.resize_if_needed(terminal_size);
421                current_frame.resize_if_needed(terminal_size);
422            }
423            RenderState::Rendered(last_rendered_frame) => {
424                last_rendered_frame.resize_if_needed(terminal_size);
425            }
426        };
427
428        terminal_size
429    }
430}

432impl<T> Drop for FrameRenderer<T>
433where
434    T: Terminal,
435{
436    fn drop(&mut self) {
437        let _unused = self.move_cursor_to_end_position();
438        let _unused = self.terminal.cursor_show();
439        let _unused = self.terminal.flush();
440    }
441}
