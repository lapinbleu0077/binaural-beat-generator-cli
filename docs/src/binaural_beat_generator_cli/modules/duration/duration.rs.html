7/// Represents common durations in minutes.
8#[derive(Debug, Clone, Copy, PartialEq)]
9pub enum Duration {
10    FiveMinutes,
11    TenMinutes,
12    FifteenMinutes,
13    TwentyMinutes,
14    ThirtyMinutes,
15    ThirtyFiveMinutes,
16    FortyMinutes,
17    FiftyMinutes,
18    SixtyMinutes,
19}
20
21/// This formatter will return the number of minutes for the given duration enum.
22impl fmt::Display for Duration {
23    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
24        match self {
25            Duration::FiveMinutes => write!(f, "5 min"),
26            Duration::TenMinutes => write!(f, "10 min"),
27            Duration::FifteenMinutes => write!(f, "15 min"),
28            Duration::TwentyMinutes => write!(f, "20 min"),
29            Duration::ThirtyMinutes => write!(f, "30 min"),
30            Duration::ThirtyFiveMinutes => write!(f, "35 min"),
31            Duration::FortyMinutes => write!(f, "40 min"),
32            Duration::FiftyMinutes => write!(f, "50 min"),
33            Duration::SixtyMinutes => write!(f, "60 min"),
34        }
35    }
36}
37
38/// This implementation will convert the duration type of minutes into integer minutes.
39impl ToMinutes for Duration {
40    fn to_minutes(&self) -> u32 {
41        match self {
42            Duration::FiveMinutes => 5,
43            Duration::TenMinutes => 10,
44            Duration::FifteenMinutes => 15,
45            Duration::TwentyMinutes => 20,
46            Duration::ThirtyMinutes => 30,
47            Duration::ThirtyFiveMinutes => 35,
48            Duration::FortyMinutes => 40,
49            Duration::FiftyMinutes => 50,
50            Duration::SixtyMinutes => 60,
51        }
52    }
53}
54
55/// This function will return the a vector list of all the supported durations.
56pub fn duration_list() -> Vec<Duration> {

62        Duration::ThirtyMinutes,
63        Duration::ThirtyFiveMinutes,
64        Duration::FortyMinutes,
65        Duration::FiftyMinutes,
66        Duration::SixtyMinutes,
67    ];
68}
