2//! ## Binaural beats is the idea that your brainwaves could be changed in a positive way by listening to slightly different tones in each ear.

22/// This is the entry point to the program.
23fn main() -> Result<(), Error> {
24    let preset_options = preset_list();
25    let duration_options = duration_list();
26    let preset_length = preset_options.len();
27
28    print_program_info("Binaural Beat Generator", "By Chris Horton");
29
30    let chosen_preset = Select::new("Choose a preset: ", preset_options)
31        .with_page_size(preset_length)
32        .prompt();
33
34    match chosen_preset {
35        Ok(preset) => {
36            let mut binaural_preset_options = BinauralPresetGroup::from(preset);
37
38            let starting_duration_index = duration_options
39                .iter()
40                .position(|&x| x == binaural_preset_options.duration)
41                .unwrap();
42
43            let chosen_duration = Select::new("Choose a duration: ", duration_options)
44                .with_starting_cursor(starting_duration_index)
45                .prompt();
46
47            match chosen_duration {
48                Ok(duration) => {
49                    //Get the chosen duration if it has changed.
50                    binaural_preset_options.duration = duration;
51                    run_binaural_beat(binaural_preset_options)?;
52                }
53                Err(err) => eprintln!(
54                    "There was an error choosing the duration, please try again. {}",
55                    err
56                ),
57            }
58        }
59        Err(err) => eprintln!("There was an error, please try again. {}", err),
60    }
61
62    Ok(())
63}

65/// A helper funciton that sets off the running of the binaural beat tones.
66/// It also spawns a new thread in order to watch for early completion.
67fn run_binaural_beat(preset_options: BinauralPresetGroup) -> Result<(), Error> {
68    let cancel_token = Arc::new(AtomicBool::new(false));
69    let cancel_token_clone = Arc::clone(&cancel_token);
70
71    // 2. Start a separate thread to listen for user input
72    std::thread::spawn(move || {
73        println!("Press Enter to stop playback.");
74
75        loop {
76            match event::read() {
77                Ok(Event::Key(key_event)) => {
78                    if key_event.kind == KeyEventKind::Press && key_event.code == KeyCode::Enter {
79                        cancel_token_clone.store(true, Ordering::Relaxed);
80                    }
81                }
82                Ok(_) => {} // Ignore other events
83                Err(err) => eprintln!("There was an error, please try again. {}", err),
84            }
85        }
86    });
87
88    generate_binaural_beats(preset_options, Arc::clone(&cancel_token))?;
89
90    Ok(())
91}

93/// A helper function that just prints out the program name and author.
94fn print_program_info(title: &str, author_text: &str) {
95    let max_chars = cmp::max(title.len(), author_text.len());
96    let separator = "=".repeat(max_chars + 10);
97    println!("{}", separator.blue());
98    println!("   {}   ", title.red().bold());
99    println!("   {}   ", author_text.red().bold().italic());
100    println!("{}", separator.blue());
101}
