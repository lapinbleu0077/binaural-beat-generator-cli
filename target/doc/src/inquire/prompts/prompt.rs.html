1//! Definitions of common behavior shared amongst all different prompt types.
2
3use crate::{error::InquireResult, input::InputActionResult, ui::CommonBackend, InquireError};
4
5use super::action::{Action, InnerAction};
6
7/// Represents the result of an action on the prompt.
8#[derive(Copy, Clone, Debug, PartialEq, Eq)]
9pub enum ActionResult {
10    /// The action resulted in a state change that requires the prompt to be
11    /// re-rendered.
12    NeedsRedraw,
13
14    /// The action either didn't result in a state change or the state
15    /// change does not require a redraw.
16    Clean,
17}
18
19impl ActionResult {
20    pub fn merge(self, other: Self) -> Self {
21        match (self, other) {
22            (Self::NeedsRedraw, _) | (_, Self::NeedsRedraw) => Self::NeedsRedraw,
23            (Self::Clean, Self::Clean) => Self::Clean,
24        }
25    }
26
27    /// Returns whether the action requires a redraw.
28    pub fn needs_redraw(&self) -> bool {
29        matches!(self, Self::NeedsRedraw)
30    }
31}
32
33impl From<InputActionResult> for ActionResult {
34    fn from(value: InputActionResult) -> Self {
35        if value.needs_redraw() {
36            Self::NeedsRedraw
37        } else {
38            Self::Clean
39        }
40    }
41}
42
43/// Shared behavior among all different prompt types.
44pub trait Prompt<Backend>
45where
46    Backend: CommonBackend,
47    Self: Sized,
48{
49    type Config;
50    type InnerAction: InnerAction<Config = Self::Config>;
51    type Output;
52
53    /// Prompt header rendered to the user.
54    fn message(&self) -> &str;
55
56    /// Returns the underlying settings of the prompt, used, among other
57    /// goals, to parse a key event into a prompt action.
58    ///
59    /// For example, a prompt might be configured to have vim mode enabled
60    /// or disabled, which affects how certain key events are parsed into
61    /// actions to the prompt.
62    fn config(&self) -> &Self::Config;
63
64    /// Hook called when a prompt is finished. Returns a string
65    /// to be rendered to the user as the final submission to the prompt.
66    ///
67    /// # Arguments
68    ///
69    /// * `answer` - Answer returned by the prompt.
70    fn format_answer(&self, answer: &Self::Output) -> String;
71
72    /// Hook called when a prompt is first started, before the first
73    /// draw happens.
74    fn setup(&mut self) -> InquireResult<()> {
75        Ok(())
76    }
77
78    /// Hook called when an input to cancel the prompt is triggered.
79    ///
80    /// Returns whether the prompt can be terminated.
81    fn pre_cancel(&mut self) -> InquireResult<bool> {
82        Ok(true)
83    }
84
85    /// Hook called when the user submits the answer to the prompt.
86    ///
87    /// On success, it should return `Some(ReturnType)` when the user
88    /// submission is valid and the prompt can graciously return.
89    ///
90    /// If the user submission is invalid or should be rejected for some reason,
91    /// this method should return `Ok(None)`.
92    ///
93    /// On `Err(*)`, the prompt is teared down.
94    fn submit(&mut self) -> InquireResult<Option<Self::Output>>;
95
96    /// Entrypoint for any business logic for the prompt. Returns the result
97    /// of the action. If the result is `Clean`, the prompt will
98    /// not be re-rendered.
99    ///
100    /// On the usual path, users' key presses are parsed into prompt actions,
101    /// which are then submitted to this method to be handled.
102    ///
103    /// On testing scenarios, developers might provide a stream of actions
104    /// to the prompt, which will then be submitted to this method just the same.
105    fn handle(&mut self, action: Self::InnerAction) -> InquireResult<ActionResult>;
106
107    /// Hook called for the rendering of the prompt UI.
108    ///
109    /// The implementation should **not** call neither `frame_setup` or
110    /// `frame_finish` methods of the underlying backend, as this is handled
111    /// by the top-level prompt method.
112    fn render(&self, backend: &mut Backend) -> InquireResult<()>;
113
114    /// Top-level implementation of a prompt's flow.
115    ///
116    /// This should not be reimplemented by types that implement this trait,
117    /// unless the situation really warrants it.
118    fn prompt(mut self, backend: &mut Backend) -> InquireResult<Self::Output> {
119        self.setup()?;
120
121        let mut last_handle = ActionResult::NeedsRedraw;
122        let final_answer = loop {
123            if last_handle.needs_redraw() {
124                backend.frame_setup()?;
125                self.render(backend)?;
126                backend.frame_finish(false)?;
127                last_handle = ActionResult::Clean;
128            }
129
130            let key = backend.read_key()?;
131            let action = Action::from_key(key, self.config());
132
133            if let Some(action) = action {
134                last_handle = match action {
135                    Action::Submit => {
136                        if let Some(answer) = self.submit()? {
137                            break answer;
138                        }
139                        ActionResult::NeedsRedraw
140                    }
141                    Action::Cancel => {
142                        let pre_cancel_result = self.pre_cancel()?;
143
144                        if pre_cancel_result {
145                            backend.frame_setup()?;
146                            backend.render_canceled_prompt(self.message())?;
147                            backend.frame_finish(true)?;
148                            return Err(InquireError::OperationCanceled);
149                        }
150
151                        ActionResult::NeedsRedraw
152                    }
153                    Action::Interrupt => return Err(InquireError::OperationInterrupted),
154                    Action::Inner(inner_action) => self.handle(inner_action)?,
155                };
156            }
157        };
158
159        let formatted = self.format_answer(&final_answer);
160
161        backend.frame_setup()?;
162        backend.render_prompt_with_answer(self.message(), &formatted)?;
163        backend.frame_finish(true)?;
164
165        Ok(final_answer)
166    }
167}
